RRK_oa_on_game_started = {
    effect = {
        every_country = {
            # Set starting Prime Ministers
            if = {
                limit = {
                    RRK_st_is_constitutional_monarchy = yes
                }
                RRK_se_set_prime_minister = yes
                RRK_se_update_head_of_gov_modifier = yes
            }
            
            # Mark all monarchies with the variable
            if = {
                limit = {
                    has_law = law_type:law_monarchy
                }
                set_variable = {
                    name  = RRK_preserve_monarchy
                    value = no
                }
            }
        }
    }
}

on_game_started_after_lobby = {
    on_actions = {
        RRK_oa_on_game_started
    }
}
