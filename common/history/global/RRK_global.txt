GLOBAL = {
    every_country = {
        # Set starting Prime Ministers
        # Game needs to actually be started for starting governments to be applied or something, so we do it 1 day into
        # the game.
        if = {
            limit = {
                RRK_st_is_constitutional_monarchy = yes
            }
            trigger_event = {
                id = RRK_ev.1
                days = 1
            }
        }
        
        # Mark all monarchies with the variable
        if = {
            limit = {
                has_law = law_type:law_monarchy
            }
            set_variable = {
                name  = RRK_preserve_monarchy
                value = no
            }
        }
        
        # Unmark all non-monarches - necessary because apparently there is on action processing during history set up
        # and we need to clear out the variable for any country where it was set as a result of starting laws being set
        else_if = {
            limit = {
                has_variable = RRK_preserve_monarchy
            }
            remove_variable = RRK_preserve_monarchy
        }
    }
}
