japanese_imperial_clothes_trigger = {
	exists = scope:character
	scope:character = {
		is_ruler = yes
		exists = home_country
     	home_country = {
            OR = {
                has_law = law_type:law_monarchy
                RRK_st_is_crowned_republic = yes # RRK
            }
     	}
		OR = {
			AND = {
				culture = {
					has_discrimination_trait = yamato
				}
				OR = {
					NOT = {
						exists = c:JAP
					}
					AND = {
						exists = c:JAP
						NOT = { home_country = c:JAP }
					}
				}
			}
			AND = {
				exists = c:JAP
				home_country = {
					this = c:JAP
					NOT = { has_variable = meiji_restoration_complete }
				}
			}
		}
	}
}

arabic_fez_trigger = {
	exists = scope:character
	scope:character = {
		exists = home_country
		OR = {
			home_country = {
				OR = {
					is_country_type = recognized
					is_country_type = colonial
				}	
			}
			home_country = {
				top_overlord = {
					OR = {
						is_country_type = recognized
						is_country_type = colonial
					}	
				}
			}
			home_country = {
				exists = c:EGY
				c:EGY = this
				c:EGY = {
                    OR = {
                        has_law = law_type:law_monarchy
                        RRK_st_is_crowned_republic = yes # RRK
                    }
                }
			}
		}
	}       
}
