REPLACE:japanese_imperial_clothes_trigger = {
	exists = scope:character
	scope:character = {
		is_ruler = yes
		exists = home_country
		home_country = {
			OR = {
				has_law = law_type:law_monarchy
				RRK_st_is_crowned_republic = yes # RRK
			}
		}
		OR = {
			AND = {
				culture = {
					has_discrimination_trait = heritage_japanese
				}
				OR = {
					NOT = {
						exists = c:JAP
					}
					AND = {
						exists = c:JAP
						NOT = { home_country = c:JAP }
					}
				}
			}
			AND = {
				exists = c:JAP
				home_country = {
					c:JAP ?= THIS
					NOT = { has_variable = meiji_restoration_complete }
				}
			}
		}
	}
}

REPLACE:chinese_court_clothes_trigger = {
	exists = scope:character
	scope:character = {
		NOT = {
			AND = {
				exists = home_country
				is_ruler = yes
				culture = {
					has_discrimination_trait = heritage_tai
					has_discrimination_trait_group = heritage_group_southeast_asian
				}
		     	home_country = {
		     		OR = {
			     		has_law = law_type:law_monarchy
						RRK_st_is_crowned_republic = yes # RRK
			     		has_law_or_variant = law_type:law_theocracy
			     	}
		     	}
			}
		}
		culture = {
			NOT = { this = cu:kalmyk }
			NOT = { this = cu:korean }
			OR = {
				has_discrimination_trait = tradition_sinosphere
				has_discrimination_trait = heritage_han
				has_discrimination_trait = language_tibetan
				has_discrimination_trait_group = language_group_mongolic
				has_discrimination_trait = heritage_tai
				has_discrimination_trait = language_hmong
			}
		}
	}
}

REPLACE:arabic_fez_trigger = {
	exists = scope:character
	scope:character = {
		exists = home_country
		OR = {
			home_country = {
				OR = {
					is_country_type = recognized
					is_country_type = colonial
				}
			}
			home_country = {
				top_overlord = {
					OR = {
						is_country_type = recognized
						is_country_type = colonial
					}
				}
			}
			home_country = {
				exists = c:EGY
				c:EGY = this
				c:EGY ?= {
					OR = {
						has_law = law_type:law_monarchy
						RRK_st_is_crowned_republic = yes # RRK
					}
				}
			}
		}
	}
}

REPLACE:monarchy_clothes_trigger = {
	exists = scope:character
	scope:character = {
		exists = owner
		NOT = { culture = cu:griqua } # Griqua captains wear plainer clothing
		OR = {
			AND = {
				OR = {
					is_female = yes # Female Regents wear more stylish clothes than male ones, as female regents trend towards being royals themselves
					owner ?= { has_gov_regency = no }
				}
				is_ruler = yes
			}
	     	is_heir = yes
		}
		owner = {
			OR = {
				has_law = law_type:law_monarchy
				RRK_st_is_crowned_republic = yes # RRK
			}
		}
	}
}
